解读this,call,apply,bind
js
js 2019-01-03

前言

在之前的执行上下文中，我们知道了每个函数都有一个执行上下文，对于每个执行上下文都有三个重要属性：

  • 作用域链
  • 变量对象
  • this 指向

今天我们来弄清楚 this 指向的是什么，还有 call 、 apply 、 bind 是怎么回事

this 指向

函数调用，创建了执行上下文，确认了 this 指向,所以 this 的指向完全取决于函数在哪里被调用。

首先我们看看代码观察一下 this 的指向

var a = 'a'
function foo(){
    console.log(this.a)
}
foo() //'a'

var o = {
    a:'aaa',
    fn: foo,
}
o.fn() // 'aaa'

var b = o.fn;
b() //'a'

我们可以看到不同的调用方式 this.a 的结果也不同，那么它是以什么规则来确定 this 指向的呢

作为一个函数调用

// "use strict"
var a = 'a'
function foo(){
    var a = 'aa'
    console.log(this.a)
    console.log(this)
}
foo() //'a'

foo() 在全局执行上下文中被调用，这个时候 this 默认指向 全局对象 window，但是在严格模式，会打印 undefined。其实这段代码还可以这样看，调用 foo() 就是调用 window.foo(),根据前面所说 this 的指向完全取决于函数在哪里被调用。foo() 作为 window 这个对象的方法调用，所以 this 指向 window，那么我们就可以知道下一条规则

函数作为对象方法调用

function foo(){
    var a = 'aa'
    console.log(this.a)
    console.log(this)
}
var A = {
    a:'a',
    fn:foo
}
A.fn() 
//'a'
// A{}

按照上述规则来看， this 指向 A 这个对象 ，foo 中的 this.a 就会打印出 A.a 的值，那我们在写个稍微复杂的例子，加以论证

function foo(){
    var a = 'aa'
    console.log(this.a)
    console.log(this)
}
var A = {
    a:'a',
    fn:foo,
    b:{
        fn:foo,
    }
}
A.b.fn()
// undefined
// A.b{}
Object.prototype.a = 'b'
A.b.fn()
// 'b'
// A.b{}

this 的指向很明朗，指向 A.b。它不会向父级查找属性，只会向原型链查找属性 如果对原型有问题的可以去看看原型 Prototype这篇文章加深下理解。我们还可以在进阶一下

var a = 'aaaa'
function foo(){
    function fn(){
        var a = 'aaa'
        console.log(this.a)
    }
    var a = 'aa'
    var A = {
        a:'a',
        fn:fn,
        b:{
            fn:fn,
        },
        c:this.a
    }
    return A
}
foo().fn() // 'a'
var o = foo().fn
o() // 'aaaa'
foo().b.fn() //undefined
console.log(foo().c) // 'aaaa'

分析一波：以下所述的都是非严格模式下的
foo() 返回一个 A 对象，foo().fn() 相当于 A.fn() 所以 this 指向 A。
o 只是引用 A.fn 的地址，这里主要看o在哪执行，我们看到 o 在全局上下文中执行 所以 this 指向 window。
同样的 A.b.fn() 跟上面的例子一样 指向 A.b，那么就返回 undefined.
最后只是访问 A.c 的属性值，并不是函数调用，使用 {} 是不会形成新的作用域，所以为 c 赋值时的 this.a 中的 this 指向是全局对象。

构造函数 this 指向 （new）

在构造函数中的 this 指向当前类的一个实例

var height = 180
function foo(name,age){
    this.name = name
    this.age = age,
    this.fn = function(){
        console.log(this.name + this.age)
    },
    this.getHeight = function(){
        console.log(this.height)
    }
}
var o = new foo('darkbo',18)
o.fn() // 'darkbo18'
o.getHeight() //undefined

用 new 创建的实例 o，我们可以粗略的看成这种情况

o = {
    name: 'darkbo',
    age: 18,
    fn:function(){
        console.log(this.name + this.age)
    },
    getHeight:function(){
        console.log(this.height)
    }
}

这样就很明显的看出 this 的指向,所以就能得出上述例子的结果。其实 new 真实情况是这样

  • 创建一个新对象
  • 将构造函数的作用域赋给新对象（因此 this 就指向了这个新对象）
  • 执行构造函数中的代码（为这个新对象添加属性）
  • 返回新对象

那么 this 是如何指向这个新对象的呢？那我们先来理解下 call、 apply、 bind。

call

call 是每个函数都有的一个方法，它允许你在调用函数时为函数指定上下文。也就是使用一个指定的 this 值和对应的参数前提下调用某个函数或方法

举个栗子

function foo(age,height){
    console.log(this.name + age + height)
}

var o = {
    name:'darkbo'
}
foo.call(o,18,180) //darkbo18180

其实 call 只是将 o 改造成这样：

var o = {
    name:'darkbo',
    foo:function(age,height){
        console.log(this.name + age + height)
    }
}
o.foo(18,180)

当然上述代码只是一个模拟等同的过程，不要试图去访问 o 的属性, o 还是 {name:'darkbo'},那我们可以去模拟一下 call 的实现。
首先我们要知道 call 都做了哪些工作：

  • 将函数 foo 作为 o 的一个属性
  • 执行 o.foo(参数)
  • 不改变原来的 o，那么只要再删除 o.foo

开始实现：

Function.prototype.newCall = function(context){
    context.fn = this // 通过上述this规则，this可以获取调用newCall的函数
    var args = []
    for(var i = 1; i < arguments.length; i++){
        args.push('arguments[' + i + ']')  //将参数保存在args数组中
    }
    eval('context.fn(' + args +')') //在eval中，args 自动调用 args.toString()方法
    //context.fn(arguments[1], arguments[2], ...);
    delete context.fn
}

//测试一下
function foo(age,height){
    console.log(this.name + age + height)
}

var o = {
    name:'darkbo'
}
foo.newCall(o,18,180) //darkbo18180

感觉像是完成了，别着急，我们还差点东西，看这个例子：

function foo(age,height){
    console.log(this.name + age + height)
    return {
        name:this.name,
        age:age,
        height:height
    }
}
var name = 'fengzibo'
var o = {
    name:'darkbo'
}
foo.call(null) //fengzibo undefined undefined
foo.call(o,18,180) //darkbo18180
console.log(foo.call(o,18,180))
// {
//     name:'darkbo',
//     age:18,
//     height:180
// }

当传入 this 的参数为 null 时，this 指向为 window。并且函数可以有返回值。那么我们再来完善下

Function.prototype.newCall = function(context){
    context = context || window //解决传null问题
    context.fn = this // 通过上述this规则，this可以获取调用newCall的函数
    var args = []
    for(var i = 1; i < arguments.length; i++){
        args.push('arguments[' + i + ']')  //将参数保存在args数组中
    }
    var res = eval('context.fn(' + args +')') //在eval中，args 自动调用 args.toString()方法
    //context.fn(arguments[1], arguments[2], ...);
    delete context.fn
    return res
}

基本功能都已实现，可是还有点隐患，如果对象本身就有 fn 这个方法的时候，就会出现问题
我们还可以在进阶一版，我们可以用es6编写：

Function.prototype.newCall = function(context, ...args){
    if (typeof context === 'object') {
        context = context || window //解决传null
    } else {
        context = Object.create(null) //传其他类型生成一个空对象
    }
    let fn = Symbol() //独一无二
    context[fn] = this // 通过上述this规则，this可以获取调用newCall的函数
    let res = context[fn](...args)
    delete context.fn
    return res
}

基本实现完毕，apply 也是同样的套路

apply

call 和 apply 一样，只是 call 是传多个参数， apply 是传入一个数组

Function.prototype.newApply = function(context, args){
    if (typeof context === 'object') {
        context = context || window //解决传null
    } else {
        context = Object.create(null) //传其他类型生成一个空对象
    }
    let fn = Symbol() //独一无二
    context[fn] = this // 通过上述this规则，this可以获取调用newCall的函数
    let res = context[fn](...args)
    delete context.fn
    return res
}

bind

bind() 方法会创建一个新函数。当这个新函数被调用时，bind() 的第一个参数将作为它运行时的 this，之后的一序列参数将会在传递的实参前传入作为它的参数。(来自于 MDN )

首先我们看下 bind 都实现了哪些功能

var name = 'fengzibo'
var o ={
    name:'darkbo'
}
function foo(age,sex){
    console.log(this.name + age + sex)
}

var bindO = foo.bind(o,18)
bindO('男') //darkbo18男

bind 可以分开传参，不像 apply 和 call 那样立即执行，而是返回一个改变 this 执行后的函数,

那么我们可以这么模拟 bind 的实现：

Function.prototype.newBind = function(context,...args){
    var self = this
    return function(rest){
        return self.call(context,...args,...rest)
    }
}

想要更深入理解bind的模拟实现，可以看看JavaScript深入之bind的模拟实现

接下来，我们回过头看看 new 的模拟实现

new

new 是个关键字 模拟一个 objectFactory(fn,...)形式，能够访问到 fn 构造函数中的属性，还能访问到 fn.prototype 中的属性,还可以拥有返回值

function objectFactory() {
    // 使用objectFactory的时候,把arguments,转化为数组
    var args = Array.prototype.slice.call(arguments);

    //提取第1个构建对象
    var Constructor = args.shift();

    // 创建constructor实例 instance 
    var instance = Object.create(Constructor.prototype);
    // var instance = new Object() //不使用Object.create的方式
    // instance.__proto__ = Constructor.prototype

    // 使用apply函数运行args, 把 instance 绑定到 this
    var temp = Constructor.apply(instance, args); 

    //返回对象判断 是object 还是 null 还是实例
    return (typeof temp === 'object' && temp !== null) ? temp : instance; 
}

箭头函数 this 指向

最后来说说箭头函数的指向
箭头函数没有自己的 this，前面我们说 this 的指向完全取决于函数在哪里被调用，这里要注意箭头函数的 this 不是调用的时候决定的，而是从自己的作用域链上一层继承 this。
举个栗子说明下

var age = 21
var o = {
    name:'darkbo',
    age:18,
    getName:function(){
        var name = 'fengzibo'
        var fn = () =>{
            var name = 'tony'
            console.log(this.name)
        }
        fn()
    },
    getAge:() =>{
        var age = 19
        var fn = () =>{
            var age = 20
            console.log(this.age)
        }
        fn()
    }
}
o.getName() // 'darkbo' this指向o
o.getAge() // 21 this指向window

o.getName() 的 this 指向是 o 对象，方法中的箭头函数 fn 就指向了 o.getName() 的 this 指向,而 o.getAge() 也是个箭头函数，那它就往上查找，由于 o 是个对象而非函数，所以 this 指向为 window 对象, 而其中的 fn 的 this 就取决于最外层的箭头函数。

最后还要注意一点，箭头函数中的 this 不会被 call 和 apply 改变

var oCall = {
    name: 'fengzibo'
}
var o = {
    name: 'darkbo',
    getName: function(){
        var fn = () =>{
            var name = 'tony'
            console.log(this.name)
        }
        return fn.call(oCall)
    }
}
o.getName() // 'darkbo' this还是指向o，并没有指向oCall
